\section*{Accessibility warnings: tabular demonstrations}
Equations should not contain \texttt{tabular} environments.
\[
 \begin{tabular}{cc}
  1 & 2 \\
  3 & 4
 \end{tabular}
\]
the following tabular is \emph{not} in math mode.

\begin{tabular}{cc}
 1 & 2 \\
 3 & 4
\end{tabular}

tabular in table
\begin{table}[!htb]
 \centering
 \begin{tabular}{cc}
  1 & 2 \\
  3 & 4
 \end{tabular}
\end{table}

tabular in table*
\begin{table*}[!htb]
 \centering
 \begin{tabular}{cc}
  1 & 2 \\
  3 & 4
 \end{tabular}
\end{table*}

\section*{Accessibility warnings: figure descriptions}

Figure without description
\begin{figure}[!htb]
 \centering
 \includegraphics[width=3cm]{example-image-a}
 \caption{caption}
\end{figure}

Figure \emph{with} description
\begin{figure}[!htb]
 \centering
 \includegraphics[width=3cm]{example-image-a}
 \figuredescription{the letter A}
 \caption{caption}
\end{figure}

Figure* without description
\begin{figure*}[!htb]
 \centering
 \includegraphics[width=3cm]{example-image-b}
